<h2>Introduction</h2>

### To install the packages:

```sh
npm install @maps-ui/react @maps-ui/tokens
```

### To use the packages

```js
import { Button, Box, Tooltip, TooltipProvider } from '@maps-ui/react'
import { colors, radii } from '@maps-ui/tokens'

export const MyPage = () => {
  return (
    <TooltipProvider>
      <Box as="section" css={{ padding: '55px' }}>
        <Tooltip content="Send Content">
          <Button css={{ borderRadius: radii.full, color: colors.white }}>
            Send
          </Button>
        </Tooltip>
      </Box>
    </TooltipProvider>
  )
}
```
